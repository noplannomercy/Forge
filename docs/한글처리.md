* `.hwpx` → zip + XML 파싱
* `.hwp` → `pyhwp2md` 실행
* 하나의 `/convert` API로 통합

## 🚀 app.py (그대로 복붙 가능)

```python
from fastapi import FastAPI, UploadFile, File, HTTPException
import zipfile, os, shutil, tempfile, subprocess
from lxml import etree

app = FastAPI()


# ---------------------------
# HWPX → MD
# ---------------------------
def hwpx_to_md(file_path: str) -> str:
    tmp_dir = tempfile.mkdtemp()

    try:
        with zipfile.ZipFile(file_path, 'r') as z:
            z.extractall(tmp_dir)

        section_path = os.path.join(tmp_dir, "Contents", "section0.xml")
        if not os.path.exists(section_path):
            return ""

        tree = etree.parse(section_path)
        root = tree.getroot()

        texts = []
        for elem in root.iter():
            if elem.text and elem.text.strip():
                texts.append(elem.text.strip())

        return "\n\n".join(texts)

    finally:
        shutil.rmtree(tmp_dir, ignore_errors=True)


# ---------------------------
# HWP → MD (pyhwp2md)
# ---------------------------
def hwp_to_md(file_path: str) -> str:
    try:
        result = subprocess.run(
            ["pyhwp2md", file_path],
            capture_output=True,
            text=True,
            timeout=30
        )
        if result.returncode != 0:
            raise Exception(result.stderr)
        return result.stdout
    except Exception as e:
        raise RuntimeError(f"HWP convert error: {e}")


# ---------------------------
# API
# ---------------------------
@app.post("/convert")
async def convert(file: UploadFile = File(...)):
    suffix = os.path.splitext(file.filename)[-1].lower()

    with tempfile.NamedTemporaryFile(delete=False, suffix=suffix) as tmp:
        tmp.write(await file.read())
        tmp_path = tmp.name

    try:
        if suffix == ".hwpx":
            md = hwpx_to_md(tmp_path)
        elif suffix == ".hwp":
            md = hwp_to_md(tmp_path)
        else:
            raise HTTPException(400, "Unsupported format")

        return {
            "text": md,
            "format": "md",
            "source_format": suffix
        }

    finally:
        os.remove(tmp_path)
```
